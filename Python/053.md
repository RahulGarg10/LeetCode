## My Algorithm
Time Complexity: $O(n)$

Space Complexity: $O(1)$
``` python3
def maxSubArray(self, nums: List[int]) -> int:
      lsum, csum = 0, 0
      count = 0
      max_num = nums[0]
      n = len(nums)
      for number in nums:
          count = count + 1
          if number > 0:
              lsum = number
              csum = number
              break
          elif number >= max_num:
              max_num = number
              lsum = max_num

      i = count
      while(i < n):
          #print(i)
          csum = csum + nums[i]
          if lsum < csum:
              lsum = csum
          if csum >= 0:
              i=i+1
              continue
          elif i == n-1:
              break
          elif nums[i+1] > 0 and lsum < nums[i+1]:
              lsum = nums[i+1]
              csum = nums[i+1]
              i=i+1
          elif  nums[i+1] > 0 and csum <= 0:
              csum = nums[i+1]
              i=i+1

          else:
              pass
          i=i+1
      return lsum

```
