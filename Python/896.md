# 896. Monotonic Array
Time Complexity: $O(n)$

Space Complexity: $O(1)$

```python3
class Solution:
    def isMonotonic(self, nums: List[int]) -> bool:
        n = len(nums)
        flag_1 = 0
        flag_2 = 0

        for i in range(n-1):
            if nums[i] < nums[i+1]: flag_1 = 1
            if nums[i] > nums[i+1]: flag_2 = 1
            if flag_1 and flag_2 : return False

        return True
```

```python3
class Solution:
    def isMonotonic(self, nums: List[int]) -> bool:
        n = len(nums)
        first = nums[0]
        last = nums[n-1]

        if first > last:
            for i in range(n-1):
                if nums[i] < nums[i+1]: return False

        elif first < last:
            for i in range(n-1):
                if nums[i] > nums[i+1]: return False

        else:
            for i in range(n-1):
                if nums[i] != nums[i+1]: return False

        return True
```
